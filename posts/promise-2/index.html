    如何优雅地实现Promise API
    June 16, 2025

    • JavaScript

    书接上回，我们实现了一个基本的Promise，这回让我们来探究Promise.all、Promise.race和Promise.allSettled的实现。

    Promise.all#

    首先明确一下目标：

    1. 接收一个数组（或可迭代对象），元素可以是值或 Promise
    2. 所有 Promise 都成功，返回一个按原始顺序组成的结果数组
    3. 只要有一个失败（reject），整个返回的 Promise 立即 reject
    4. 返回的数组顺序必须与传入顺序一致。
    5. 如果传入的数组为空，立即resolve（这一点和Promise.race有所区别，如果Promise.race传入空数组，会一直挂起）

    实现：

     1class MyPromise {
 2  // 基础实现请参考上篇
 3  static all(iterable) {
 4    // 构造一个Promise
 5    let p = new MyPromise(function () {});
 6
 7    let res = []; // 用于存放resolved value
 8    let ret = iterable.length;
 9    iterable.forEach((item, i) => {
10      MyPromise.resolve(item)
11        .then((value) => {
12          res[i] = value;
13          if (!--ret) {
14            p.settlePromise($resolved, res);
15          }
16        })
17        .catch((reason) => {
18          // 任何一个promise reject都直接reject这个Promise.all
19          p.settlePromise($rejected, reason);
20        });
21    });
22    if (!ret) {
23      p.settlePromise($resolved, []);
24    }
25    return p;
26  }
27}

    Promise.race#

    目标：

    1. 和Promise.all一样接收一个数组，数组元素可以是值可以是Promise
    2. 返回一个新的 Promise，传入的数组中谁先settle（无论resolved还是rejected）了，立即resolve/reject这个promise，其他Promise 继续执行
    3. 传入空数组时挂起这个Promise.race（既不resolve，也不reject）

    实现：

     1class MyPromise {
 2  static race(iterable) {
 3    const p = new MyPromise(function () {});
 4    const onResolved = function (val) {
 5      p.settlePromise($resolved, val);
 6    };
 7    const onRejected = function (reason) {
 8      p.settlePromise($rejected, reason);
 9    };
10    try {
11      iterable.forEach((item) => {
12        MyPromise.resolve(item).then(onResolved, onRejected);
13      });
14    } catch (err) {
15      p.settlePromise($rejected, err);
16    }
17    return p;
18  }
19}

    Promise.allSettled#

    目标：

    1. 接收一个数组，数组元素可以是普通值可以是Promise
    2. 返回新的Promise，并等待所有Promise settle（无论resolve还是reject）后resolve这个promise
    3. resolve的值由这样的对象组成，代表每个promise的结果
    {
  status, // 'fulfilled' | 'rejected'，表示每个promise的最终状态
  value?, // fulfilled value，只有fulfilled时存在
  reason?, // rejected reason，只有rejected时存在
}

    实现：

     1class MyPromise {
 2  static allSettled(iterable) {
 3    const p = new MyPromise(function () {});
 4    let res = [],
 5      ret = iterable.length;
 6
 7    try {
 8      iterable.forEach((item, i) => {
 9        MyPromise.resolve(item).then(
10          function (value) {
11            res[i] = { status: "fulfilled", value };
12            if (!--ret) p.settlePromise($resolved, res);
13          },
14          function (reason) {
15            res[i] = { status: "rejected", reason };
16            if (!--ret) p.settlePromise($resolved, res);
17          }
18        );
19      });
20    } catch (err) {
21      p.settlePromise($rejected, err);
22    }
23
24    if (!ret) p.settlePromise($resolved, []);
25
26    return p;
27  }
28}

    可以看到，基于我们已有的Promise实现，上面的三个方法其实都很简单，重要的是理解Promise 的运作机制，看到这，相信你不仅会用Promise，并且对原理也已经了然于胸了。
